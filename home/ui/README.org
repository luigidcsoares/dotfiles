#+author: Luigi D. C. Soares

Here goes any aspect of the ui to be configure at the system level. We

- Set up and install the default fonts, and configure font scaling
- Set up catppuccin theme for GTK (deprecated, but...)

#+begin_src nix :tangle default.nix
{ config, pkgs, ... }: {
  home.packages = [
    (pkgs.iosevka-bin.override { variant = "Aile"; })
    (pkgs.iosevka-bin.override { variant = "Etoile"; })
    pkgs.nerd-fonts.iosevka
    pkgs.nerd-fonts.symbols-only
  ];

  home.sessionVariables = {
    GDK_SCALE = "2";
    GDK_DPI_SCALE = "2";
    GTK_SCALE = "2";
    QT_SCALE_FACTOR = "2";
  };

  fonts.fontconfig = {
    enable = true;
    defaultFonts = {
      emoji = [ "Symbols Nerd Font Mono" ];
      monospace = [ "Iosevka Nerd Font" ];
      serif = [ "Iosevka Etoile" ];
      sansSerif = [ "Iosevka Aile" ];
    };
  };
  
  gtk = {
    enable = true;
    theme = {
      package = pkgs.magnetic-catppuccin-gtk.override {
        tweaks = [ "frappe" ];
      };
      name = "Catppuccin-GTK-Dark-Frappe";
    };
  };
  
  qt = {
    enable = true;
    platformTheme.name = "kvantum";
    style.name = "kvantum";
  };

}
#+end_src
